#!/usr/bin/env bash set -Eeuo pipefail trap 'cleanup' ERR declare e loc m pkg_mgr="unknown" loc=$(wget -qO- "https://developers.cloudflare.com/cdn-cgi/trace" | sed -n 's/^loc=//p') case "${loc^^}" in CN) echo "zh-CN" ;; TW) echo "zh-TW" ;; *) echo "en" ;; esac for m in apt-get apk dnf opkg pacman yum zypper; do command -v "$m" &>/dev/null && pkg_mgr="$m" && break done function cleanup() { local e=$? ((e != 0)) && exit "$e" } function detect_environment() { ((BASH_VERSINFO[0] > 4)) || ((BASH_VERSINFO[0] == 4 && BASH_VERSINFO[1] >= 4)) || return 1 [[ ${pkg_mgr,,} == "apt-get" ]] && dpkg --compare-versions "$(apt-get --version | head -n1 | awk '{print $2}')" ge "1.9" && pkg_mgr="apt" sed -i "s/BASH=1/BASH=0/" "${HOME}/utilkit.sh" sed -i "s/PKG_MGR=\"unknown\"/PKG_MGR=\"${pkg_mgr}\"/" "${HOME}/utilkit.sh" } function download_utilkit_sh() { if wget -qO "${HOME}/utilkit.sh" "https://raw.githubusercontent.com/OG-Open-Source/UtilKit/main/sh/localized/${lang}/utilkit.sh" 2>/dev/null; then printf "%b\n" "\x1b[0;32mDownloaded pre-localized version for ${lang}\x1b[0m" else printf "%b\n" "\x1b[0;33mPre-localized version not available, downloading default version...\x1b[0m" if ! wget -qO "${HOME}/utilkit.sh" "https://raw.githubusercontent.com/OG-Open-Source/UtilKit/main/sh/utilkit.sh"; then printf "%b\n" "\x1b[0;31mFailed to download UtilKit.sh\x1b[0m" exit 1 fi fi } lang="${1:-$(DetectLang)}" if [[ -f ${HOME}/utilkit.sh ]]; then printf "%b\n" "\x1b[0;32mUpdating UtilKit.sh...\x1b[0m" download_utilkit_sh detect_environment printf "%b\n" "\x1b[0;32mUtilKit.sh has been updated successfully\x1b[0m" else if ! crontab -l 2>/dev/null | grep -q "get_utilkit.sh"; then (crontab -l 2>/dev/null || true) | { cat printf "%b\n" "0 0 * * 0 wget -qO- https://raw.githubusercontent.com/OG-Open-Source/UtilKit/main/sh/get_utilkit.sh | bash -s -- ${lang}" } | crontab - printf "%b\n" "\x1b[0;32mAdded weekly auto-update to crontab\x1b[0m" fi printf "%b\n" "\x1b[0;32mDownloading UtilKit.sh...\x1b[0m" download_utilkit_sh detect_environment if ! grep -q "source ~/utilkit.sh" ~/.bashrc; then cat <<<"source ~/utilkit.sh" >>~/.bashrc printf "%b\n" "\x1b[0;32mAdded source command to ~/.bashrc\x1b[0m" fi printf "%b\n" "\x1b[0;32mUtilKit.sh has been installed successfully\x1b[0m" printf "%b\n" "\x1b[0;32mPlease run 'source ~/utilkit.sh' to use it in the current session\x1b[0m" fi
